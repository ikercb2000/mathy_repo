Linn=function(para){
  cQ1=para[1]    
  cQ2=para[2]   
  cQ3=para[3]   
  cR=para[4]  
  cQ=diag(c(cQ1,cQ2,cQ3,0,0))
  kf=Kfilter(y,A,mu0,Sigma0,Phi,cQ,cR)
  return(kf$like)
}
